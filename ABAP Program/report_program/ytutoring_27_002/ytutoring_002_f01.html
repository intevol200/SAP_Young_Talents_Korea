FORM get_flight_info CHANGING lt_info LIKE gt_info
                              lt_scarr LIKE gt_scarr
                              lt_sfli LIKE gt_sfli
                              lt_spfli LIKE gt_spfli.
  SELECT *
    FROM spfli
    INTO CORRESPONDING FIELDS OF TABLE lt_info.

  SELECT carrid carrname
    FROM scarr
    INTO CORRESPONDING FIELDS OF TABLE lt_scarr.

  SELECT carrid planetype
    FROM sflight
    INTO CORRESPONDING FIELDS OF TABLE lt_sfli.

  SELECT carrid connid fltype
    FROM spfli
    INTO CORRESPONDING FIELDS OF TABLE lt_spfli.
ENDFORM.
*&---------------------------------------------------------------------*
*& Form final_info
*&---------------------------------------------------------------------*
*&---------------------------------------------------------------------*
FORM final_info USING VALUE(lt_scarr) LIKE gt_scarr
                      VALUE(ls_scarr) LIKE gs_scarr
                      VALUE(lt_sfli) LIKE gt_sfli
                      VALUE(ls_sfli) LIKE gs_sfli
                      VALUE(lt_spfli) LIKE gt_spfli
                      VALUE(ls_spfli) LIKE gs_spfli
                CHANGING lt_info LIKE gt_info
                         ls_info LIKE gs_info.
  LOOP AT lt_info INTO ls_info.
    READ TABLE lt_scarr INTO ls_scarr WITH KEY carrid = ls_info-carrid.
    ls_info-carrname = ls_scarr-carrname.

    READ TABLE lt_sfli INTO ls_sfli WITH KEY carrid = ls_info-carrid.
    ls_info-planetype = ls_sfli-planetype.

    READ TABLE lt_spfli INTO ls_spfli WITH KEY carrid = ls_info-carrid
                                               connid = ls_info-connid.
    ls_info-fltype = ls_spfli-fltype.

    PERFORM get_fltype_domain CHANGING ls_info.
    PERFORM check_status CHANGING ls_info.

    MODIFY lt_info FROM ls_info.
  ENDLOOP.
ENDFORM.
*&---------------------------------------------------------------------*
*& Form check_status
*&---------------------------------------------------------------------*
FORM check_status CHANGING ls_info_n LIKE gs_info.
  IF ls_info_n-planetype = 'A380-800' OR
     ls_info_n-planetype = 'A319-100'.

    ls_info_n-check = 'Confirm'.

  ELSE.
    ls_info_n-check = 'Delayed'.
  ENDIF.
ENDFORM.
