REPORT ytutoring_27_001.

TYPES: BEGIN OF str_fli,
         carrid   TYPE sflight-carrid,
         carrname TYPE scarr-carrname,
         connid   TYPE sflight-connid,
         fldate   TYPE sflight-fldate,
         price    TYPE sflight-price,
       END OF str_fli.

DATA: gs_fli TYPE str_fli,
      gt_fli TYPE TABLE OF str_fli.

PARAMETERS: pa_id LIKE gs_fli-carrid,
            pa_con LIKE gs_fli-connid.

START-OF-SELECTION.
  SELECT a~carrid a~carrname
         b~connid
         c~fldate c~price
    FROM scarr AS a JOIN spfli AS b
    ON a~carrid = b~carrid
    JOIN sflight AS c
    ON a~carrid = c~carrid
    INTO CORRESPONDING FIELDS OF TABLE gt_fli
    WHERE a~carrid = pa_id
    AND b~connid = pa_con.

  cl_demo_output=>display_data( gt_fli ).
