FORM find_data USING VALUE(la_car)
                      VALUE(la_con)
                      VALUE(la_data)
                CHANGING ls_flight.

SELECT SINGLE *
  FROM sflight
  INTO CORRESPONDING FIELDS OF ls_flight
  WHERE carrid = la_car
   AND connid = la_con
  AND fldate = la_data.

ENDFORM.

*& Form find_carr
*&---------------------------------------------------------------------*
FORM find_carr USING VALUE(la_car)
                CHANGING ls_carrier.

SELECT SINGLE *
  FROM scarr
  INTO CORRESPONDING FIELDS OF ls_carrier
  WHERE carrid = la_car.

ENDFORM.
