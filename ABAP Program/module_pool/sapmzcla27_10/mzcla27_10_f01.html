FORM set_default .
*  gs_book-carrid = 'AA'.
*  gs_book-connid = '0017'.
*  gs_book-fldate = '20220112'.
*  gs_book-bookid = '3637'.

  SELECT carrid connid fldate bookid
    FROM sbook UP TO 1 ROWS
    INTO CORRESPONDING FIELDS OF gs_book
    WHERE carrid = 'AA'
    AND connid = '0017'
    AND fldate = '20220112'
    AND bookid = '3637'.
  ENDSELECT.
ENDFORM.

*& Form get_booking_info
*&---------------------------------------------------------------------*
FORM get_booking_info .
  SELECT SINGLE *
    FROM sbook AS a JOIN scarr AS b
    ON a~carrid = b~carrid
    INTO CORRESPONDING FIELDS OF zsa27_11
    WHERE a~carrid = gs_book-carrid
    AND a~connid = gs_book-connid
    AND a~fldate = gs_book-fldate
    AND a~bookid = gs_book-bookid.

  IF sy-subrc IS NOT INITIAL.
    MESSAGE i016(pn) WITH 'Data not Found.'.
    RETURN.
  ENDIF.
ENDFORM.
