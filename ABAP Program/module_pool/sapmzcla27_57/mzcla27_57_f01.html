FORM set_default_condition USING VALUE(p_code)
                            CHANGING ps_cond LIKE gs_cond.
  CLEAR gs_cond.
  SELECT SINGLE carrid connid fldate
    FROM sflight
    INTO CORRESPONDING FIELDS OF ps_cond
   WHERE carrid = p_code.

ENDFORM.
