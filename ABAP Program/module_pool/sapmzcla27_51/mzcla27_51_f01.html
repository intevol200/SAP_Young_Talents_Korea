FORM get_flight_info USING VALUE(p_carrid)
                               VALUE(p_connid)
                               VALUE(p_fldate)
                      CHANGING p_subrc.
  CLEAR p_subrc.
  CLEAR sflight.
  SELECT SINGLE *
    FROM sflight
    INTO CORRESPONDING FIELDS OF sflight
*    INTO sflight
   WHERE carrid = p_carrid
     AND connid = p_connid
     AND fldate = p_fldate.
  IF sy-subrc <> 0.
    p_subrc = 4.
    CLEAR sflight.
    RETURN.
  ENDIF.
ENDFORM.
*&---------------------------------------------------------------------*
*& Form init_condition
*&---------------------------------------------------------------------*
FORM init_condition .
  IF gs_cond IS INITIAL.
    SELECT carrid connid fldate
      FROM sflight UP TO 1 ROWS
      INTO CORRESPONDING FIELDS OF gs_cond.
    ENDSELECT.
  ENDIF.
ENDFORM.
