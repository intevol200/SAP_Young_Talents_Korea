MODULE user_command_0100 INPUT.
  CLEAR zsa21_03.

  MOVE-CORRESPONDING zsa21_02 TO gs_regi.

  SELECT SINGLE *
    INTO CORRESPONDING FIELDS OF zsa21_03
    FROM zinfo_regi AS a
    JOIN zinfo_student AS b
    ON a~stdid = b~stdid
    JOIN zinfo_subject_t AS c
    ON a~subcode = c~subcode
    WHERE a~stdid = gs_regi-stdid
    AND a~subcode = gs_regi-subcode
    AND a~regiyear = gs_regi-regiyear
    AND a~regisemi = gs_regi-regisemi.
ENDMODULE.
