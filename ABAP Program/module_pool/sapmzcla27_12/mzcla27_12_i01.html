  CLEAR zsa27_11.

  MOVE-CORRESPONDING zsa27_10 TO gs_book.
  MESSAGE i016(pn) WITH 'Screen 100:' sy-ucomm ok_code.

  DATA: gv_carrname TYPE scarr-carrname.

  CASE ok_code.

      MESSAGE s016(pn) WITH 'Call Screen 200'.
      PERFORM get_booking_info.

      gv_carrname = zsa27_11-carrname.

*      SET SCREEN 200.

  ENDCASE.
ENDMODULE.
