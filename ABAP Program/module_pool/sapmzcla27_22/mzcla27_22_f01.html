FORM get_customer_liest .
  SELECT *
    FROM scustom
    INTO CORRESPONDING FIELDS OF TABLE gt_info
    WHERE id BETWEEN gv_cid_b AND gv_cid_e.

  IF sy-subrc <> 0.
    MESSAGE i016(pn) WITH 'Data Not Found'.
  ENDIF.
